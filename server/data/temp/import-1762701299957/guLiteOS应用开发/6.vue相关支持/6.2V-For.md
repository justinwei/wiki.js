layout 循环

```
... layout.xml
<div>
    <label :text="title" />
    <label v-for="item in list" :text="item.name" />
    <div  v-for="item in list">
        <label :text="item.name" />
        <label :text="|" />
        <label :text="item.sex" />
    </div>
</div>

....script.js
page = Vue({
    data:{
        "title":"hello world",
        "content":{
            "text":"txt_content"
        }
        "list":[
            {"name":"justin", "sex":"male"},
            {"name","alice", "sex":"female"}
        ]
    }
});    
... 显示结果如下：
hello world
justin
alice

justin|male
alice|female
```
