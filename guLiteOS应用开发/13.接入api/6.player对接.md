# 一、对接的api介绍

```
/*
    创建播放器
*/
int player_wrapper_create(player_record_t* info);
```
```
/*
    停止播放器
*/
int player_wrapper_stop(player_record_t* info);
```
```
/*
    注销播放器
*/
int player_wrapper_destroy(player_record_t* info);
```
```
/*
    获取播放器状态
*/
int player_wrapper_get_status(player_record_t* info);
```
```
/*
    获取播放器播放音频的偏移量
*/
int player_wrapper_get_curr_offset(player_record_t* info);
```
```
/*
    获取播放器播放音频的总大小
*/
int player_wrapper_get_size(player_record_t* info);
```
```
/*
    seek播放器播放的音频到指定位置
*/
int player_wrapper_seek(player_record_t* info);
```

# 二、形参结构

```
typedef enum {
	MEDIA_PLAYER_NOT_PAUSED = 0,
	MEDIA_PLAYER_PAUSED,
} player_pause_status_t;
```

```
typedef struct {
	unsigned int play_offset;
	unsigned int total_offset;
	char url[MEDIA_URL_LEN + 1];
        char name[MEDIA_NAME_LEN + 1];
	unsigned int stop_statue;
	unsigned char block;
} base_play_info_t;
```
```
typedef struct {
	player_pause_status_t paused;
	base_play_info_t curr_info;
} player_record_t;
```
