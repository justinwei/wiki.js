# 一、对接的api介绍

```
/*
    切换模式，sta/ap
*/
int wifi_wrapper_switch_mode(e_wifi_mode_t mode);
```
```
/*
    失能能wifi
*/
int wifi_wrapper_disable();
```
```
/*
    使能wifi，开始链接
*/
int wifi_wrapper_enable(char *ssid, int ssid_len, char *pwd);
```
```
/*
    
*/
int wifi_wrapper_connect();
```
```
/*
    扫描ssid的最大值
*/
int wifi_wrapper_sta_bss_max_count(unsigned char size);
```
```
/*
    扫描时间间隔/持续时间
*/
int wifi_wrapper_sta_scan_interval(int sec);
```
```
/*
    触发一次扫描
*/
int wifi_wrapper_sta_scan_once(void);
```
```
/*
    获取扫描到的结果
*/
int wifi_wrapper_sta_scan_result(wifi_wrapper_scan_results_t *results);
```
```
/*
    获取WiFi当前状态
*/
unsigned int wifi_wrapper_event_subtype(unsigned int event);
```
```
/*
    初始化wifi
*/
int wifi_wrapper_init(void (*cb)(unsigned int event, unsigned int data, void *arg));
```
```
/*
    获取当前连接的wifi的信号值
*/
int wifi_wrapper_get_cur_wifi_signal();
```
```
/*
    断开WiFi连接
*/
int wifi_wrapper_disconnect();
```
