绑定属性时，在属性名称前加":",例如 
绑定的写变量的json路径
```
... layout.xml
<div>
    <label :text="title">
    <label :text="content.text">
    <label :text="list[0].name">        
</div>

....script.js
/*Json 绑定数据描述
   在sript.js的data属性里
*/
page = Vue({
    data:{
        "title":"hello world",
        "content":{
            "text":"txt_content"
        }
        "list":[
            {"name":"justin"},
            {"name","alice"}
        ]    
    }
})
{

}

... 显示结果如下：
hello world
txt_content
justin
```
