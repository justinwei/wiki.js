# 一、对接的api介绍

```
typedef struct {
	void *data;    /*数据指针。需要使用realize_unit_mem_malloc()分配*/
	unsigned int data_len;    /*数据长度*/
	unsigned int last_data_state;    /*是否是后一包数据*/
} pic_data_t;
```
```
typedef struct {
	int (*open)(void);        /*笔头按下需要调用的回调*/
	int (*write)(pic_data_t* pic_data);    /*有数据需要触发的回调*/
	int (*close)(void);        /*笔头抬起需要触发的回调*/
} camera_cb_t;
```
```
/*
    获取回调函数，用于驱动触发时调用回调通知上层
*/
camera_cb_t *camera_wrapper_get_cb(void);
```
```
/*
    初始化摄像头
*/
int camera_wrapper_init(camera_cb_t *cb);
```
```
/*
    注销/关闭摄像头
*/
int camera_wrapper_deinit(void);
```
