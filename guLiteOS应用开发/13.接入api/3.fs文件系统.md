# 一、对接的api介绍
* 特殊平台需要自行填充，linux、 freertos、 macos平台可使用默认配置，不需要修改

```
////////////////////////////////////////////////////////////////////////////
/*****************************fs file api**********************************/
////////////////////////////////////////////////////////////////////////////

/**
 * @brief 打开文件
 *
 * @param path 文件路径
 * @param fp 打开模式
 * @return 文件句柄
 */
void* fs_wrapper_open(const char *path, file_open_flags_t flags);
```
```
/**
 * @brief 关闭文件
 *
 * @param 文件句柄
 * @return int 0 succ
 */
int fs_wrapper_close(void *fp);
```
```
/**
 * @brief 读文件
 *
 * @param fp 文件句柄
 * @param buf 文件内容指针
 * @param btr 需要读取的字节数
 * @param br 实际读到的字节数
 * @return int 0 succ
 */
int fs_wrapper_read(void *fp, void *buf, unsigned int btr, unsigned int *br);
```
```
/**
 * @brief 读文件
 *
 * @param fp 文件句柄
 * @param buf 文件内容指针
 * @param btw 需要写入的字节数
 * @param bw 实际写入的字节数
 * @return int 0 succ
 */
int fs_wrapper_write(void *fp, void *buf, unsigned int btw, unsigned int *bw);
```
```
/**
 * @brief 定位到文件的指定位置
 *
 * @param fp 文件句柄
 * @param pos 想要偏移的位置
 * @param whence 偏移的起始终位置
 * @return int 0 succ
 */
int fs_wrapper_seek( void *fp, unsigned int pos, flie_whence_flags_t whence);
```
```
/**
 * @brief 获取当前文件指针位置
 *
 * @param fp 文件句柄
 * @param pos_p 文件指针位置
 * @return int 0 succ
 */
int fs_wrapper_tell( void *fp, unsigned int *pos_p);
```
```
/**
 * @brief 执行同步操作
 *
 * @param fp 文件句柄
 * @param path_new 新名称
 * @return int 0 succ
 */
int fs_wrapper_sync(void* fp);
```
```
/**
 * @brief 判断是否到文件末尾
 *
 * @param fp 文件句柄
 * @return int 1 文件末尾
 */
int fs_wrapper_eof(void* fp);
```
```
////////////////////////////////////////////////////////////////////////////
/*****************************fs dir api***********************************/
////////////////////////////////////////////////////////////////////////////
/**
 * @brief 打开一个文件夹
 *
 * @param path 文件路径
 * @return 文件夹描述句柄
 */
void* fs_wrapper_opendir(const char *path);
```
```
/**
 * @brief 读取文件夹
 *
 * @param dir_p 文件夹描述句柄
 * @param info 获取文件夹的信息
 * @return int 0 succ
 */
int fs_wrapper_readdir(void *dir_p, fs_wrapper_info_t *info);
```
```
/**
 * @brief 关闭文件夹
 *
 * @param dir_p 文件夹描述句柄
 * @return int 0 succ
 */
int fs_wrapper_closedir(void *dir_p);
```
```
/**
 * @brief 新建文件夹
 *
 * @param dir_p 文件夹描述句柄
 * @return int 0 succ
 */
int fs_wrapper_mkdir(const char* path);
```
```
////////////////////////////////////////////////////////////////////////////
/******************************other api***********************************/
////////////////////////////////////////////////////////////////////////////

/**
 * @brief 获取文件或者文件夹的状态以及信息
 *
 * @param path 路径
 * @param info 获取到的信息
 * @return int 0 succ
 */
int fs_wrapper_stat(const char *path, fs_wrapper_info_t *info);
```
```
/**
 * @brief 重命名文件或者文件夹
 *
 * @param path_old 旧名称
 * @param path_new 新名称
 * @return int 0 succ
 */
int fs_wrapper_rename(const char* path_old, const char* path_new);
```
```
/**
 * @brief 删除文件或者文件夹
 *
 * @param path 路径
 * @param path_new 新名称
 * @return int 0 succ
 */
int fs_wrapper_remove(const char* path);
```
```
/**
 * @brief 挂载文件系统
 *  //目前只支持tf卡与lfs-falsh
 * @param path 要挂载的路径（fatfs表示tf卡, lfs表示flash user, lfs_ota标识flash ota）
 * @return int 0 succ
 */
int fs_wrapper_mount(const char* path);
```
```
/**
 * @brief 卸载文件系统
 *  //目前只支持tf卡与lfs-falsh
 * @param path 要卸载的路径（fatfs表示tf卡, lfs表示flash user, lfs_ota标识flash ota）
 * @return int 0 succ
 */
int fs_wrapper_unmount(const char* path);
```
```
/**
 * @brief 获取文件系统挂载状态
 *  //目前只支持tf卡与lfs-falsh
 * @param void
 * @return int 所有文件系统的挂载状态
 */
int fs_wrapper_get_mount_state(void);
```
```
/**
 * @brief 获取文件系统挂载的实际空间
 *  //目前只支持tf卡与lfs-falsh
 * @param space_info 实际空间的输出信息
 * @return int 0 succ
 */
int fs_wrapper_space(const char* path, fs_space_t *space_info);
```
```
/**
 * @brief 格式化磁盘
 *  //目前只支持tf卡与lfs-flash
 * @param space_info 实际空间的输出信息
 * @return int 0 succ
 */
int fs_wrapper_format(const char* path);
```
